## 競技ルール
　募集要項 (https://www.procon.gr.jp/wp-content/uploads/2022/04/7cd7daa1edc6ad3c7077207f1814c2fc-1.pdf#page=9) に書かれています。読みデータの読みの開始位置をずらしたものが複数重ね合わせられたものが問題データとなっています。

## おことわり
　この問題設定が、本当に運営が想定していたものと一致するのか、与えられた情報が表しているものであるのかには決定的な報告はないそうです。配布された問題サンプルに添付されている音源の形式をもとに作成したつもりですが、もしかしたら勘違いしたままだったりするのかもしれません(というかそもそも大した順位ではなかったので色々と微妙な可能性が高いです)。

## スクリプト
```python
import json
import random
import pydub
from pydub import AudioSegment


def to_audio_segment(file_name: str, file_format: str) -> AudioSegment | None:
    try:
        audio_segment = AudioSegment.from_file(file_name, format=file_format)
        return audio_segment
    except FileNotFoundError as e:
        print(e)

    return None


def slice_audio(audio: AudioSegment | None, start: int, end: int) -> AudioSegment:
    if audio is None:
        return AudioSegment.empty()

    ## [start, end) に切り出し
    return audio[start*1000:end*1000]

def merge_audios(audios: list[AudioSegment]) -> AudioSegment:
    ## 先頭のデータに以降のデータを重ねる(AudioSegment.overlay())
    merged_audio = audios.pop()
    for audio in audios:
        merged_audio = merged_audio.overlay(audio)

    return merged_audio


def main():
    ## 音源のメタデータ(json)
    questions_metadata = {}

    ## 一度に複数個作成
    for question_number in range(20):
        ## 重ね合わせる個数
        audio_count = random.randint(8, 20)
        ## 1~44のデータから重ね合わせる個数分無作為に選択
        audio_numbers = random.sample(range(1, 44 + 1), audio_count)

        audio_segments = []
        for audio_number in audio_numbers:
            audio_segment = to_audio_segment(f'../JKspeech/J{audio_number:02}.wav', 'wav')

            ## 開始位置・終了を適当に指定
            start = random.randint(0, 2)
            end = random.randint(start + 4, start + 6)
            audio_segment = slice_audio(audio_segment, start, end)

            audio_segments.append(audio_segment)

        ## 作成したサンプルデータのエクスポート, メタデータの書き込み
        question_id = f'Q{question_number+1:03}_audio_numbera'
        merged_audio = merge_audios(audio_segments)
        merged_audio.export(f'../questions_ja/{question_id}.wav', format='wav')
        questions_metadata.update({question_id: {
            'count': audio_count,
            'correct_numbers': sorted(audio_numbers)
        }})


    with open('../questions_ja/metadata.json', 'w') as f:
        json.dump(questions_metadata, f, ensure_ascii=False, indent=2)

if __name__ == '__main__':
    main()
    print("Done")
```
